attribute vec2 aVertexPosition;
attribute vec4 aVertexColor;

uniform vec2 uResolution;
uniform float uZIndex;

varying lowp vec4 vColor;

void main(void) {
  vec2 vRatio = aVertexPosition / uResolution;
  vec2 clipSpace = (vRatio * 2.0) - 1.0;
  gl_Position = vec4(clipSpace * vec2(1, -1), uZIndex, 1);
  vColor = aVertexColor;
}
